#include <Servo.h>

Servo myservo;
int inches = 0;
int cm = 0;

long readUltrasonicDistance(int triggerPin, int echoPin) {
    pinMode(triggerPin, OUTPUT);
    digitalWrite(triggerPin, LOW);
    delayMicroseconds(2);
    digitalWrite(triggerPin, HIGH);
    delayMicroseconds(10);
    digitalWrite(triggerPin, LOW);
    pinMode(triggerPin, INPUT);
    return pulseIn(echoPin, HIGH);
}

void setup() {
    myservo.attach(9);
    Serial.begin(9600);
}

void loop() {
    cm = 0.01723 * readUltrasonicDistance(7, 7);
    if (cm < 100) {
        myservo.write(90);
        delay(10);
    } else if (cm > 100) {
        myservo.write(0);
        delay(100);
    }
    inches = (cm / 2.54);
    Serial.print(inches);
    Serial.print("in, ");
    Serial.print(cm);
    Serial.println("cm");
    delay(100);
}
